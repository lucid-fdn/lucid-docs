### lucidInstallModule

Install an ERC-7579 module (policy, payout, or receipt) on a smart account. The module type and initialization data must be provided in the request body.


#### Example Usage


```typescript
import { RaijinLabsLucidAi } from "raijin-labs-lucid-ai";

const raijinLabsLucidAi = new RaijinLabsLucidAi();

async function run() {
  const result = await raijinLabsLucidAi.modules.lucidInstallModule({
    chainId: "<id>",
    account: "81158790",
    moduleType: "<value>",
    moduleAddress: "<value>",
  });

  console.log(result);
}

run();
```

### lucidConfigurePayoutModule

```typescript

    chainId: "<id>",
    account: "86222423",
    recipients: [
      "<value 1>",
      "<value 2>",
    ],
    basisPoints: [
      122328,
    ],
  });

}

run();
```
